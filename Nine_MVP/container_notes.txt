to build a artifact repository:

gcloud artifacts repositories create ninebatchcontainer \
    --repository-format=docker \
    --location=us-central1
	
to build container image:

gcloud builds submit \
  -t us-central1-docker.pkg.dev/nine-quality-test/ninebatchcontainer/batch-chunk-and-embedding:v1.00  BatchJobs/
